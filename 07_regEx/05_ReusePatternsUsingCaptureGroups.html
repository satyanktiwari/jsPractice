                      let repeatStr = "row row row your boat";
                      let repeatRegex = /(\w+) \1 \1/;
                      repeatRegex.test(repeatStr); // Returns true
                      repeatStr.match(repeatRegex); // Returns ["row row row", "row"];
                      
                      in the code above
                      \1
                      1st grp matched is stored in a temp variable
                      \1 indicates use the same group for matching
                      \1 indicates a space then use the same group for match
                      \1 \1 indicates repeat the first match twice 
                      This can be written as below
                      \s\1\s\1
                      \s is whitespace shorthand
